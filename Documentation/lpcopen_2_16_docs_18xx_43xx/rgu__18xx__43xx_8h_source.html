119 STATIC INLINE void Chip_RGU_TriggerReset(CHIP_RGU_RST_T ResetNumber)
120 {
121  LPC_RGU->RESET_CTRL[ResetNumber >> 5] = 1 << (ResetNumber & 31);
122  /* Reset will auto clear after 1 clock cycle */
123 }

130 STATIC INLINE bool Chip_RGU_InReset(CHIP_RGU_RST_T ResetNumber)
131 {
132  return !(LPC_RGU->RESET_ACTIVE_STATUS[ResetNumber >> 5] & (1 << (ResetNumber & 31)));
133 }
